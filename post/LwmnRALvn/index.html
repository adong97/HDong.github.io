为什么做生产环境监控部署包

目前我们很多部署在客户环境中的云管其实都没有配置监控，导致很多时候出现问题后，客户找到我们比较被动，还会出现磁盘满导致服务不可用。同时对于规模比较大的客户，资源数量增长比较快，也不清楚整体系统资源随资源数量增长的变化性能情况。为了能够及时了解云管的运行情况、及时发现故障并解决，制作了云管生产环境监控部署包。

使用场景

在云管生产实施后，导入部署包，把Prometheus监控部署包部署到一台监控主机，把Node Exporter部署包部署到需要监控的虚拟机，把mysql exporter部署包部署到MySQL组主机，把Cadvisor部署包部署到需要监控的docker服务虚拟机，打开监控主机登录Grafana，即可看到生产云管系统的监控情况，并能在部署时指定的告警通知邮箱中收到告警。

二、prometheus部署包使用说明：

由于proemtheus使用专为监控指标数据设计的高性能时序数据库TSDB，为保证服务器时间准确一致，部署包中安装了ntpd时间同步服务。

因此如果是内网环境，请最好配置本地yum源和时间同步服务来保证时间准确一致。

2.1 工具功能说明

用于监控云管环境中的机器状态和服务状态，确保出现异常及时告警维护。
已支持监控node节点的CPU、内存、磁盘、网络等，MySQL服务、docker容器，后续根据需求更新监控服务和监控指标。

五 一键部署脚本

5.1 监控告警一键部署脚本

#!/bin/bash

#设置环境信息
[ `whoami` != "root" ] && echo "######## Please Use root Install......" && exit 71
IP=`ip addr |awk '/inet /' |sed -n '2p' |awk -F' ' '{print $2}' |awk -F'/' '{print $1}'`
install_dir=${{Install_Dir}}      #安装目录  /opt
tool_dir=${install_dir}/software  #部署包目录 /opt/software
data_dir=${{Data_Dir}}            #prometheus数据目录  /opt
LogFile=${{logfile}}              #日志文件  /opt/install-tool.log
file_dir=${{file_Dir}}            #解压文件目录  /usr/local

#配置告警邮箱
Smtp_Mail=${{Smtp_Mail}}  #smtp邮箱+端口号，例:smtp.qq.com:465
Stmp_Username=${{Smtp_Username}}  #接收告警通知人的邮箱地址，例：xxxxxxxx@qq.com
Stmp_Password=${{Stmp_Password}}  #邮箱授权码

yum install zip expect -y

#检查文件目录是否存在
if [ ! -f ${LogFile} ];then
    touch ${LogFile}
fi

if [ ! -d ${install_dir}/Consul_Host ];then
    mkdir -p ${install_dir}/Consul_Host
else
    echo "目录已存在" >> ${LogFile}
fi

#安装前初始化，检查防火墙、selinux
function InitInstall()
{
echo "===================开始安装前初始化，检查防火墙、selinux===================" >> ${LogFile} 
#Disable SeLinux
if [ -s /etc/selinux/config ]; then
    sed -i 's/SELINUX=enforcing/SELINUX=disabled/g' /etc/selinux/config
fi
setenforce 0

echo "Selinux已关闭" >>  ${LogFile}

#关闭防火墙
systemctl status firewalld > /dev/null
if [ $? -eq 0 ];then
    systemctl stop firewalld > /dev/null 2>&1
    echo -e "firewalld关闭成功！" >> ${LogFile}
else
    echo "未开启防火墙无需添加规则！" >> ${LogFile}
fi
}

#安装ntpd
function Install_Ntpd()
{
if [ -f /usr/sbin/ntpdate ];then
    echo "the ntp-service is installed" >> ${LogFile}
else
    yum install -y ntp  && mv /etc/ntp.conf /etc/ntp.conf.bak
    cat > /etc/ntp.conf <<EOF
driftfile /var/lib/ntp/drift
restrict default kod nomodify notrap nopeer noquery
restrict -6 default kod nomodify notrap nopeer noquery
restrict 127.0.0.1
restrict -6 ::1
#server 0.centos.pool.ntp.org iburst
#server 1.centos.pool.ntp.org iburst
#server 2.centos.pool.ntp.org iburst
#server 3.centos.pool.ntp.org iburst
#定义使用的上游 ntp服务器，将原来的注释
server ntp1.aliyun.com
server time1.aliyun.com
#允许上层时间服务器主动修改本机时间
restrict time1.aliyun.com nomodify notrap noquery
restrict ntp1.aliyun.com nomodify notrap noquery

includefile /etc/ntp/crypto/pw
keys /etc/ntp/keys
EOF
    
	#启动ntpd服务
    systemctl start ntpd
    if [ $? -eq 0 ];then
        echo "ntpd启动成功" >> ${LogFile}
    else
        echo "ntpd启动失败" >> ${LogFile}
    fi
fi
}

#下载工具安装包
function Download()
{
echo "======================开始下载工具安装包=========================" >>  ${LogFile} 
cd ${install_dir}
if [ ! -f ${{software}} ];then

cd ${tool_dir}

}

#解压并启动prometheus
function Install_Prometheus()
{
echo "========================开始解压并启动prometheus...===================" >> ${LogFile}
cd ${tool_dir}
tar -zxvf prometheus-2.30.3.linux-amd64.tar.gz -C ${file_dir}
ln -sv ${file_dir}/prometheus-2.30.3.linux-amd64 ${file_dir}/prometheus
cat > /etc/systemd/system/prometheus.service <<EOF
[Unit]
Description=Prometheus
Documentation=https://prometheus.io/docs/introduction/overview/
Wants=network-online.target
After=network-online.target

[Service]
User=root
Group=root
Type=simple
# 启动脚本
ExecStart=${file_dir}/prometheus/prometheus \
    --config.file ${file_dir}/prometheus/prometheus.yml \
    --storage.tsdb.path ${data_dir}/prometheus/ \

[Install]
WantedBy=multi-user.target
EOF

chmod a+x /etc/systemd/system/prometheus.service
systemctl daemon-reload && systemctl start prometheus.service && systemctl enable prometheus.service
if [ $? -eq 0 ];then
    echo -e "prometheus启动成功..." >> ${LogFile}
else
    echo "prometheus启动失败" >> ${LogFile}
	exit
fi
}


#安装consul
function Install_Consul()
{
echo "======================开始安装consul=====================" >> ${LogFile}
if [ ! -d ${data_dir}/consul ];then
    mkdir -pv /etc/consul.d  && mkdir -pv ${data_dir}/consul/shell
else
    echo "consul目录已存在" >> ${Logfile}
fi

cd ${tool_dir}
unzip consul_1.10.3_linux_amd64.zip && mv consul /usr/bin/


#配置启动脚本
cat > /etc/consul.d/service.json <<EOF
{
"server": true,
"ui":true,
"datacenter": "consul",
"data_dir": "${data_dir}/consul",
"node_name": "consul-node",
"bootstrap_expect":1,
"client_addr": "0.0.0.0",
"bind_addr": "${IP}"
}
EOF

cat > /etc/systemd/system/consul.service<<EOF
[Unit]
Description=consul-service
After=network.target

[Service]
Type=simple
User=root
ExecStart=/usr/bin/consul agent -config-dir /etc/consul.d
Restart=on-failure

[Install]
WantedBy=multi-user.target
EOF

#启动consul服务
systemctl daemon-reload  && systemctl enable consul  && systemctl start consul
sleep 10
netstat -anp | grep consul
if [ $? -eq 0 ];then
    echo "consul已启动" >> ${LogFile}
else
    echo "consul未启动" >> ${LogFile}
fi

#将consul添加到prometheus.yml
cat >> ${file_dir}/prometheus/prometheus.yml <<EOF
  # consul
  - job_name: "consul-prometheus"
    consul_sd_configs:
      - server: ${IP}:8500
    relabel_configs:
    - source_labels: [__meta_consul_service]
      action: drop
      regex: consul
EOF

#重启prometheus
systemctl restart prometheus
sleep 10
}


function Install_Alter()
{
echo "============================开始安装Alertmanager=====================" >> ${LogFile}
#解压文件
tar -zxvf ${tool_dir}/alertmanager-0.23.0.linux-amd64.tar.gz -C ${file_dir}/
ln -sv  ${file_dir}/alertmanager-0.23.0.linux-amd64  ${file_dir}/alertmanager

#修改alertmanager配置文件
cd ${file_dir}/alertmanager
mv alertmanager.yml alertmanager.yml.bak
cat > alertmanager-email.yml <<EOF
global:
  resolve_timeout: 10s
  smtp_smarthost: "${Smtp_Mail}"
  smtp_from: "${Stmp_Username}"
  smtp_auth_username: "${Stmp_Username}"
  smtp_auth_password: "${Stmp_Password}"
  smtp_require_tls: false

route:
  group_by: ['altername']
  group_wait: 10s
  group_interval: 20m
  repeat_interval: 1h
  receiver: 'email'

receivers:
- name: 'email'
  email_configs:
    - to: "${Stmp_Username}"

templates:
  - "/usr/local/alertmanager/template/email.tmpl"

inhibit_rules:
  - source_match:
      severity: 'critical'
    target_match:
      severity: 'warning'
    equal: ['alertname', 'dev', 'instance']
EOF


#检查alertmanager配置文件
cd ${file_dir}/alertmanager
./amtool check-config alertmanager-email.yml  >> ${LogFile}
if [ $? -eq 0 ];then
    echo "Checking 'alertmanager-email.yml'  SUCCESS" >> ${LogFile}
else
    echo "Checking 'alertmanager-email.yml'  FAILED" >> ${LogFile}
fi

#添加告警模板
mkdir -p ${file_dir}/alertmanager/template
cp ${tool_dir}/rules/email.tmpl  ${file_dir}/alertmanager/template/

# 加入开机启动
cat > /etc/systemd/system/alertmanager.service <<EOF
[Unit]
Description=alertmanager
After=local-fs.target network-online.target network.target
Wants=local-fs.target network-online.target network.target
 
[Service]
ExecStart=${file_dir}/alertmanager/alertmanager --config.file=${file_dir}/alertmanager/alertmanager-email.yml
Restart=on-failure

[Install]
WantedBy=multi-user.target
EOF

#启动服务
systemctl daemon-reload  && systemctl enable alertmanager  && systemctl start alertmanager
netstat -nap|grep alertmanager
if [ $? -eq 0 ];then
    echo "alertmanager服务已启动" >> ${LogFile}
else
    echo "alertmanager服务启动失败" >> ${LogFile}
fi
}

function Config_Alert()
{
#Prometheus配置
#进入prometheus安装目录，编辑prometheus.yml,开启alertmanager监控
sed -i "s/# - alertmanager:9093/  - ${IP}:9093/g"  ${file_dir}/prometheus/prometheus.yml 

#开启告警配置
sed -i '15a\    - "rules/*.yml"' ${file_dir}/prometheus/prometheus.yml

#监控Alertmanager
cat >> ${file_dir}/prometheus/prometheus.yml << EOF

  # alertmanager
  - job_name: 'alertmanager'
    static_configs:
      - targets: ["localhost:9093"]
EOF

#检查并重新加载配置文件
cd ${file_dir}/prometheus
./promtool check config prometheus.yml  >> ${LogFile}
if [ $? -eq 0 ];then
    echo "Checking prometheus.yml  SUCCESS" >> ${LogFile}
else
    echo "Checking prometheus.yml FAILED" >> ${LogFile}
fi

#创建告警目录&添加报警规则
#在prometheus.yml的同级目录下，创建rules目录，在该rules目录下创建node_alerts.yml文件，内容如下：
mkdir -p  ${file_dir}/prometheus/rules
cp ${tool_dir}/rules/*.yml ${file_dir}/prometheus/rules/

#重启Prometheus
systemctl restart prometheus
systemctl restart alertmanager

netstat -nap|grep alertmanager
if [ $? -eq 0 ];then
    echo "alertmanager服务已重启完成" >> ${LogFile}
else
    echo "alertmanager服务重启失败" >> ${LogFile}
fi

netstat -nap|grep prometheus
if [ $? -eq 0 ];then

driftfile /var/lib/ntp/drift

restrict -6 default kod nomodify notrap nopeer noquery
restrict 127.0.0.1
restrict -6 ::1
#server 0.centos.pool.ntp.org iburst
#server 1.centos.pool.ntp.org iburst
#server 2.centos.pool.ntp.org iburst

#允许上层时间服务器主动修改本机时间

restrict ntp1.aliyun.com nomodify notrap noquery

includefile /etc/ntp/crypto/pw
keys /etc/ntp/keys

	#启动ntpd服务
